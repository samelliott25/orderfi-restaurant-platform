Build Kitchen Display System (KDS): Real-time page for orders. Use WebSockets for updates. Example code:tsx

// client/src/pages/kds.tsx
import { useEffect, useState } from 'react';
import { io } from 'socket.io-client';
import { Card, CardContent, CardHeader } from '@/components/ui/card';
import { Badge } from '@/components/ui/badge';

interface Order { id: string; items: { name: string; quantity: number }[]; status: 'pending' | 'preparing' | 'ready'; table: string; timestamp: Date; }

const KDS = () => {
  const [orders, setOrders] = useState<Order[]>([]);
  const socket = io('http://localhost:3000'); // Backend URL

  useEffect(() => {
    socket.on('new-order', (order: Order) => setOrders(prev => [...prev, order]));
    socket.on('update-order', (updated: Order) => setOrders(prev => prev.map(o => o.id === updated.id ? updated : o)));
    return () => socket.disconnect();
  }, []);

  const updateStatus = (id: string, status: Order['status']) => socket.emit('update-status', { id, status });

  return (
    <div className="grid grid-cols-1 md:grid-cols-3 gap-4 p-4">
      {orders.map(order => (
        <Card key={order.id} className="glassmorphism">
          <CardHeader>
            <div className="flex justify-between">
              <span>Order #{order.id} - Table {order.table}</span>
              <Badge variant={order.status === 'ready' ? 'success' : 'default'}>{order.status}</Badge>
            </div>
            <span>{new Date(order.timestamp).toLocaleTimeString()}</span>
          </CardHeader>
          <CardContent>
            <ul>{order.items.map((item, i) => <li key={i}>{item.quantity}x {item.name}</li>)}</ul>
            <div className="mt-4 flex gap-2">
              <button onClick={() => updateStatus(order.id, 'preparing')}>Prepare</button>
              <button onClick={() => updateStatus(order.id, 'ready')}>Ready</button>
            </div>
          </CardContent>
        </Card>
      ))}
    </div>
  );
};
export default KDS;
