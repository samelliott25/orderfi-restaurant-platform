/*––––– VARIABLES –––––*/
:root {
  --orb-size:       300px;
  --spin-duration:  60s;
  --sheen-duration: 6s;

  --c1: #FF8A00;  /* orange */
  --c2: #FF1493;  /* deep pink */
  --c3: #800080;  /* purple */

  --sheen-color: rgba(255,255,255,0.5);
}

/*––––– CENTER & BACKDROP –––––*/
body {
  margin: 0;
  height: 100vh;
  display: flex;
  align-items: center;
  justify-content: center;
  background: #000;
  perspective: 800px;
}

/*––––– ORB BASE –––––*/
.orb {
  position: relative;
  width:  var(--orb-size);
  height: var(--orb-size);
  border-radius: 50%;
  overflow: hidden;

  /* radial gradient: orange → pink → purple */
  background: radial-gradient(
    circle at 30% 30%,
    var(--c1) 0%,
    var(--c2) 50%,
    var(--c3) 100%
  );

  /* marble veins filter + blend */
  filter: url(#iridescentNoise);
  -webkit-filter: url(#iridescentNoise);
  background-blend-mode: overlay;
  -webkit-background-blend-mode: overlay;

  /* depth shadows */
  box-shadow:
    inset 0 0 60px rgba(0,0,0,0.4),
          0 20px 50px rgba(0,0,0,0.6);

  /* 3D spin */
  transform-style: preserve-3d;
  animation: spin var(--spin-duration) linear infinite;
  -webkit-animation: spin var(--spin-duration) linear infinite;
}

/*––––– GLOSSY SHEEN –––––*/
.orb::after {
  content: '';
  position: absolute;
  top: -15%;
  left: 10%;
  width: 120%;
  height: 120%;
  background: radial-gradient(
    circle at top left,
    var(--sheen-color) 0%,
    transparent 60%
  );
  animation: sheen var(--sheen-duration) ease-in-out infinite;
  -webkit-animation: sheen var(--sheen-duration) ease-in-out infinite;
}
